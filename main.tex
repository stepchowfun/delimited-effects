      \medskip

      \begin{prooftree}
          \AxiomC{}
        \RightLabel{(\textsc{T-Constant})}
        \UnaryInfC{$\tjudgment{\ccontext}{\econst}{\tx{\xempty}{\tconst}}$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\tanno{\evar}{\twithx} \in \ccontext$}
        \RightLabel{(\textsc{T-Variable})}
        \UnaryInfC{$\tjudgment{\ccontext}{\evar}{\twithx}$}
      \end{prooftree}

      \begin{prooftree}
        \AxiomC{$\tjudgment{\cextend{\ccontext}{\tanno{\evar}{\ttype}}}{\eterm}{\twithx}$}
        \RightLabel{(\textsc{T-Abstraction})}
        \UnaryInfC{$\tjudgment{\ccontext}{\eabs{\tanno{\evar}{\ttype}}{\eterm}}{\tarrow{\ttype}{\twithx}}$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\tjudgment{\ccontext}{\eterm_1}{\tx{\xeffects_1}{\ttype_1}}$}
          \AxiomC{$\tjudgment{\ccontext}{\eterm_2}{\tx{\xeffects_2}{\parens{\tarrow{\ttype_1}{\tx{\xeffects_3}{\ttype_2}}}}}$}
        \RightLabel{(\textsc{T-Application})}
        \BinaryInfC{$\tjudgment{\ccontext}{\eapp{\eterm_2}{\eterm_1}}{\tx{\xunion{\xeffects_1}{\xunion{\xeffects_2}{\xeffects_3}}}{\ttype_2}}$}
      \end{prooftree}

      \begin{prooftree}
          \AxiomC{$\tjudgment{\ccontext}{\eterm}{\tx{\xeffects}{\ttype}}$}
        \RightLabel{(\textsc{T-Weaken})}
        \UnaryInfC{$\tjudgment{\ccontext}{\eterm}{\tx{\xextend{\xeffects}{\xeffect}}{\ttype}}$}
      \end{prooftree}

      \begin{prooftree}
        \AxiomC{\Shortstack[c]{
          {$\pall = \left\{\pitem{\pname_1}{\eterm_1}, \ldots, \pitem{\pname_n}{\eterm_n}\right\}$}
          {$\tjudgment{\ccontext}{\eterm_i}{\tx{\xeffects_i}{\ttype_i}}$}
          {$\twithx_i = \tx{\xextend{\xeffects_i}{\xeffect}}{\ttype_i}$}
          {$\xmaplookup{\xeffect} = \left\{\pitem{\pname_1}{\twithx_1}, \ldots, \pitem{\pname_n}{\twithx_n}\right\}$}
          {$\tjudgment{\cextend{\cextend{\cextend{\ccontext}{\tanno{\pname_1}{\twithx_1}}}{\ldots}}{\tanno{\pname_n}{\twithx_n}}}{\eterm}{\tx{\xeffects}{\ttype}}$}}
        }
        \RightLabel{(\textsc{T-Provide})}
        \UnaryInfC{$\tjudgment{\ccontext}{\eprovide{\pall}{\xeffect}{\eterm}}{\tx{\xeffects - \xeffect}{\ttype}}$}
      \end{prooftree}
